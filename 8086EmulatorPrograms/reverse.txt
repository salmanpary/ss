MESSAGE1 DB 'THE ORIGINAL STRING IS'
STRING1 DB 'MBADRI'
 DB CR, LF, '$'
STRLENGTH EQU $-STRING1
MESSAGE2 DB 'THE REVERSED STRING IS'
STRING2 DB 5 DUP(' ')
 DB CR, LF, '$'
DATA ENDS
CODE SEGMENT
ASSUMECS:CODE, DS: DATA
START: MOV AX, DATA
MOV DS, AX
MOV ES,AX
LEA SI,STRING1
LEA DI, STRING2+STRLENGTH-1
MOV CX, STRLENGTH
TOP: CLD
LODSB
 STD
STOSB
 LOOP TOP
MOV AH,09H
LEA DX, MESSAGE1
INT 21H
MOV AH,09H
LEA DX, MESSAGE2
INT 21H
QUIT: MOV AL,0
MOV AH, 04CH
INT 21H
CODE ENDS
END START
